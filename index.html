Firebase Operations Setup

This website has complete Firebase SDK modular integrations inside. Set up your Firebase project console and paste your configuration credentials in the dedicated javascript section.

firestore.rules 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Helper function
    function isAdmin() {
      return request.auth != null && request.auth.token.email.matches('.*@svwoodworks\\.com');
    }
 
    match /artifacts/{appId}/public/data/{collectionName}/{document} {
      allow read: if true;
      allow write: if isAdmin();
    }
  }
}
storage.rules 
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Helper function
    function isAdmin() {
      return request.auth != null && request.auth.token.email.matches('.*@svwoodworks\\.com');
    }
 
    match /{allPaths=**} {
      allow read: if true;
      allow write: if isAdmin();
    }
  }
}
